import React, { useState, useEffect, useRef } from 'react'

interface SubSection {
  id: string
  title: string
  html?: string
  text?: string
}

interface Part {
  part: string
  title: string
  subs: SubSection[]
}

interface CommandPaletteProps {
  parts: Part[]
  glossary: Record<string, string>
  onSelectSection: (partIdx: number, secIdx: number) => void
  onToggleTheme: () => void
  onToggleFocus: () => void
  isOpen: boolean
  onClose: () => void
}

export default function CommandPalette({ parts, glossary, onSelectSection, onToggleTheme, onToggleFocus, isOpen, onClose }: CommandPaletteProps) {
  const [query, setQuery] = useState('')
  const [results, setResults] = useState<{ type: 'section' | 'glossary' | 'action'; label?: string; title: string; desc?: string; pIdx?: number; sIdx?: number; id?: string }[]>([])
  const inputRef = useRef<HTMLInputElement>(null)

  // System Actions (Always available or matched)
  const actions = [
    { type: 'action' as const, title: 'Toggle Dark / Light Theme', desc: 'Switch between vintage bookstore and sleek dark mode', handler: onToggleTheme },
    { type: 'action' as const, title: 'Toggle Focus Mode', desc: 'Hide sidebars and read with zero distraction', handler: onToggleFocus },
    { type: 'action' as const, title: 'Reset Reading Progress', desc: 'Clear local history and restart reading logs', handler: () => { if(confirm("Reset progress?")) { localStorage.removeItem('pmn-read'); window.location.reload(); } } },
  ]

  useEffect(() => {
    if (isOpen) {
      setQuery('')
      setResults([])
      setTimeout(() => inputRef.current?.focus(), 50)
    }
  }, [isOpen])

  useEffect(() => {
    const term = query.toLowerCase().trim()
    if (!term) {
      setResults([])
      return
    }

    const matches: typeof results = []

    // 1. Match Actions
    actions.forEach(a => {
      if (a.title.toLowerCase().includes(term)) matches.push(a)
    })

    // 2. Match Glossary
    Object.entries(glossary).forEach(([t, def]) => {
      if (t.toLowerCase().includes(term)) {
        matches.push({ type: 'glossary', title: t, desc: def })
      }
    })

    // 3. Match Sections
    parts.forEach((p, pIdx) => {
      p.subs.forEach((s, sIdx) => {
        if (s.title.toLowerCase().includes(term) || s.id.toLowerCase().includes(term)) {
          matches.push({ type: 'section', label: p.part, title: s.title, id: s.id, pIdx, sIdx })
        }
      })
    })

    setResults(matches.slice(0, 10))
  }, [query, parts, glossary])

  useEffect(() => {
    const handleKeyDown = (e: KeyboardEvent) => {
      if (e.key === 'Escape') onClose()
    }
    window.addEventListener('keydown', handleKeyDown)
    return () => window.removeEventListener('keydown', handleKeyDown)
  }, [onClose])

  if (!isOpen) return null

  return (
    <div className="fixed inset-0 z-[500] flex items-start justify-center pt-24 px-4 bg-black/60 backdrop-blur-md select-none transition-opacity duration-200">
      <div className="w-full max-w-[580px] bg-pmn-bg border border-pmn-rule shadow-[0_20px_50px_rgba(0,0,0,0.4)] overflow-hidden flex flex-col max-h-[500px] rounded-xs">
        <div className="flex items-center gap-4 border-b border-pmn-rule px-5 py-4 bg-pmn-bg2/50">
          <span className="font-pmn-mono text-lg text-pmn-acc opacity-80">&#8981;</span>
          <input
            ref={inputRef}
            type="text"
            className="flex-1 bg-transparent border-none outline-none font-pmn-mono text-[0.88rem] text-pmn-ink placeholder:text-pmn-mute/30"
            placeholder="Type a command, section, or keyword..."
            value={query}
            onChange={e => setQuery(e.target.value)}
          />
          <div className="flex items-center gap-2">
            <span className="font-pmn-mono text-[0.55rem] text-pmn-mute/60 border border-pmn-rule px-1.5 py-0.5 rounded-xs uppercase">Esc</span>
          </div>
        </div>

        <div className="flex-1 overflow-y-auto custom-scrollbar">
          {!query.trim() ? (
            <div className="p-8 flex flex-col gap-6">
              <span className="font-pmn-mono text-[0.6rem] text-pmn-mute uppercase tracking-[0.2em] font-bold">Suggested Actions</span>
              <div className="flex flex-col gap-1">
                {actions.map((a, i) => (
                  <button key={i} onClick={() => { (a as any).handler(); onClose(); }} className="w-full text-left p-4 hover:bg-pmn-acc/5 rounded-xs group flex justify-between items-center transition-all">
                    <div>
                      <h4 className="font-pmn-head text-[0.95rem] text-pmn-ink font-bold">{a.title}</h4>
                      <p className="font-pmn-mono text-[0.65rem] text-pmn-mute">{a.desc}</p>
                    </div>
                    <span className="font-pmn-mono text-[0.55rem] text-pmn-acc opacity-0 group-hover:opacity-100 transition-opacity uppercase tracking-widest">Execute</span>
                  </button>
                ))}
              </div>
            </div>
          ) : results.length === 0 ? (
            <div className="p-12 text-center text-pmn-mute font-pmn-body italic opacity-60">
              No matching modules or system commands for "{query}".
            </div>
          ) : (
            <div className="divide-y divide-pmn-rule/40">
              {results.map((res, i) => (
                <button
                  key={i}
                  onClick={() => {
                    if (res.type === 'action') (res as any).handler();
                    else if (res.type === 'section') onSelectSection(res.pIdx!, res.sIdx!);
                    else if (res.type === 'glossary') { /* handle glossary jump if needed or just search */ }
                    onClose()
                  }}
                  className="w-full text-left p-5 hover:bg-pmn-acc/5 flex flex-col gap-1 cursor-pointer transition-all group"
                >
                  <div className="flex justify-between items-baseline font-pmn-mono text-[0.58rem] tracking-[0.15em] text-pmn-acc uppercase font-bold">
                    <span>{res.type === 'section' ? `Part ${res.label} \u00B7 ${res.id}` : res.type}</span>
                    <span className="text-pmn-mute/40 group-hover:text-pmn-acc transition-colors">{res.type === 'action' ? 'Execute' : 'Navigate'} &rarr;</span>
                  </div>
                  <h4 className="font-pmn-head text-[1rem] font-bold text-pmn-ink group-hover:text-pmn-acc transition-colors leading-tight">{res.title}</h4>
                  {res.desc && <p className="font-pmn-body text-[0.8rem] text-pmn-mute italic line-clamp-1 opacity-70">{res.desc}</p>}
                </button>
              ))}
            </div>
          )}
        </div>
      </div>
    </div>
  )
}
